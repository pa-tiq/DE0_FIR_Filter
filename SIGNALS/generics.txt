Wmult			: INTEGER 	:= 16		;-- Multiplier bit width 2*Win
Wadd 			: INTEGER 	:= 27		;-- Adder width = Wmult+log2(L)-1
Wout 			: INTEGER 	:= 10		;-- Output bit width: between Win and Wadd

RANGE_HIGH 		: INTEGER 	:= 127		; --must change pattern too
LFilter  		: INTEGER 	:= 4096		); -- Filter length
